    <script>
        const socket = io();
        
        function joinRoom() {
            const roomId = document.getElementById('roomId').value;
            const playerName = document.getElementById('playerName').value;
            socket.emit('joinRoom', { roomId, playerName });
            document.getElementById('gameScreen').style.display = 'flex';
        }

        function startGame() {
            const roomId = document.getElementById('roomId').value;
            socket.emit('startGame', roomId);
        }

        // 初始化 5x5 地圖與筆記本 UI
        function initUI() {
            const board = document.getElementById('kitchenBoard');
            board.innerHTML = '';
            for (let i = 0; i < 25; i++) {
                const tile = document.createElement('div');
                tile.classList.add('tile');
                if (i === 12) {
                    tile.classList.add('center-tile');
                    tile.innerText = '🍳\n中央烤爐';
                } else {
                    tile.innerText = '🧱\n流理台';
                }
                board.appendChild(tile);
            }

            const notebook = document.getElementById('notebookGrid');
            notebook.innerHTML = '';
            for (let i = 1; i <= 10; i++) {
                const item = document.createElement('div');
                item.classList.add('note-item');
                item.innerText = `配方 ${i.toString().padStart(2, '0')} ⭕`;
                item.onclick = () => {
                    item.innerText = item.innerText.includes('⭕') ? `配方 ${i.toString().padStart(2, '0')} ❌` : `配方 ${i.toString().padStart(2, '0')} ⭕`;
                };
                notebook.appendChild(item);
            }
        }

        socket.on('gameStarted', (data) => {
            alert('遊戲開始！AI 已補滿，手牌已暗中分發。');
            initUI();
        });
    </script>
